144%----------------------------------------------
145function [in, on] = vec_inpolygon(Nv,x,y,xv,yv)
146% vectorize the computation.
147
148
149
150Np = length(x);
151x = x(ones(Nv,1),:);
152y = y(ones(Nv,1),:);
153
154% Compute scale factors for eps that are based on the original vertex 
155% locations. This ensures that the test points that lie on the boundary 
156% will be evaluated using an appropriately scaled tolerance.
157% (m and mp1 will be reused for setting up adjacent vertices later on.)
158m = 1:Nv-1;
159mp1 = 2:Nv;
160avx = abs(0.5*(  xv(m,:) + xv(mp1,:)));
161avy = abs(0.5*(yv(m,:)+yv(mp1,:)));
162scaleFactor = max(avx(m), avy(m));
163scaleFactor = max(scaleFactor, avx(m,:).*avy(m,:) );
164% Translate the vertices so that the test points are
165% at the origin.
166xv = xv(:,ones(1,Np)) - x;
167yv = yv(:,ones(1,Np)) - y;
168
169% Compute the quadrant number for the vertices relative
170% to the test points.
171posX = xv > 0;
172posY = yv > 0;
173negX = ~posX;
174negY = ~posY;
175quad = (negX & posY) + 2*(negX & negY) + ...
176    3*(posX & negY);
177
178% Ignore crossings between distinct edge loops that are separated by NaNs
179nanidx = isnan(xv) | isnan(yv);
180quad(nanidx) = NaN;
181% Compute the sign() of the cross product and dot product
182% of adjacent vertices.
183theCrossProd = xv(m,:) .* yv(mp1,:) - xv(mp1,:) .* yv(m,:);
184signCrossProduct = sign(theCrossProd);
185
186
187% Adjust values that are within epsilon of the polygon boundary.
188% Making epsilon larger will treat points close to the boundary as 
189% being "on" the boundary. A factor of 3 was found from experiment to be
190% a good margin to hedge against roundoff.
191scaledEps = scaleFactor*eps*3;
192idx = bsxfun(@lt, abs(theCrossProd), scaledEps);
193signCrossProduct(idx) = 0;
194
195dotProduct = xv(m,:) .* xv(mp1,:) + yv(m,:) .* yv(mp1,:);
196
197% Compute the vertex quadrant changes for each test point.
198diffQuad = diff(quad);
199
200% Fix up the quadrant differences.  Replace 3 by -1 and -3 by 1.
201% Any quadrant difference with an absolute value of 2 should have
202% the same sign as the cross product.
203idx = (abs(diffQuad) == 3);
204diffQuad(idx) = -diffQuad(idx)/3;
205idx = (abs(diffQuad) == 2);
206diffQuad(idx) = 2*signCrossProduct(idx);
207
208% Find the inside points.
209% Ignore crossings between distinct loops that are separated by NaNs
210nanidx = isnan(diffQuad);
211diffQuad(nanidx) = 0;
212in = (sum(diffQuad) ~= 0);
213
214% Find the points on the polygon.  If the cross product is 0 and
215% the dot product is nonpositive anywhere, then the corresponding
216% point must be on the contour.
217on = any((signCrossProduct == 0) & (dotProduct <= 0));
218
219in = in | on;
220
221
