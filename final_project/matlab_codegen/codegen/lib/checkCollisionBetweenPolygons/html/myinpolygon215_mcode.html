1function [in, on] = myinpolygon(x,y,xv,yv) %#codegen
  2%INPOLYGON True for points inside or on a polygonal region.
  3%   IN = INPOLYGON(X,Y,XV,YV) returns a matrix IN the size of X and Y.
  4%   IN(p,q) = 1 if the point (X(p,q), Y(p,q)) is either strictly inside or
  5%   on the edge of the polygonal region whose vertices are specified by the
  6%   vectors XV and YV; otherwise IN(p,q) = 0.
  7%
  8%   [IN ON] = INPOLYGON(X,Y,XV,YV) returns a second matrix, ON, which is 
  9%   the size of X and Y.  ON(p,q) = 1 if the point (X(p,q), Y(p,q)) is on 
 10%   the edge of the polygonal region; otherwise ON(p,q) = 0.
 11%
 12%   INPOLYGON supports non-convex and self-intersecting polygons.
 13%   The function also supports multiply-connected or disjoint polygons; 
 14%   however, the distinct edge loops should be separated by NaNs. In the
 15%   case of multiply-connected polygons, the external and internal loops
 16%   should have opposite orientations; for example, a counterclockwise 
 17%   outer loop and clockwise inner loops or vice versa. Self-intersections
 18%   are not supported in this context due to the ambiguity associated with
 19%   loop orientations.
 20%
 21%   Example 1:
 22%       % Self-intersecting polygon
 23%       xv = rand(6,1); yv = rand(6,1);
 24%       xv = [xv ; xv(1)]; yv = [yv ; yv(1)];
 25%       x = rand(1000,1); y = rand(1000,1);
 26%       in = inpolygon(x,y,xv,yv);
 27%       plot(xv,yv,x(in),y(in),'.r',x(~in),y(~in),'.b')
 28%
 29%   Example 2:
 30%       % Multiply-connected polygon - a square with a square hole.
 31%       % Counterclockwise outer loop, clockwise inner loop.
 32%       xv = [0 3 3 0 0 NaN 1 1 2 2 1];
 33%       yv = [0 0 3 3 0 NaN 1 2 2 1 1];
 34%       x = rand(1000,1)*3; y = rand(1000,1)*3;
 35%       in = inpolygon(x,y,xv,yv);
 36%       plot(xv,yv,x(in),y(in),'.r',x(~in),y(~in),'.b')
 37%
 38%   Class support for inputs X,Y,XV,YV:
 39%      float: double, single
 40%
 41%   See also, DelaunayTri
 42%
 43%   Copyright 1984-2011 The MathWorks, Inc.
 44%   $Revision: 1.14.4.11 $  $Date: 2011/05/17 02:32:25 $
 45
 46% The algorithm is similar to that described in the following reference.
 47%
 48% @article{514556,
 49% author = {Kai Hormann and Alexander Agathos},
 50% title = {The point in polygon problem for arbitrary polygons},
 51% journal = {Comput. Geom. Theory Appl.},
 52% volume = {20},
 53% number = {3},
 54% year = {2001},
 55% issn = {0925-7721},
 56% pages = {131--144},
 57% doi = {http://dx.doi.org/10.1016/S0925-7721(01)00012-8},
 58% publisher = {Elsevier Science Publishers B. V.},
 59% address = {Amsterdam, The Netherlands, The Netherlands},
 60% }
 61
 62% The edge loops defining each contour are checked for closure, and if
 63% necessary they are closed.
 64
 65
 66
 67xv = xv(:);
 68yv = yv(:);
 69
 70inputSize = size(x);
 71
 72x = x(:).';
 73y = y(:).';
 74
 75mask = (x >= min(xv)) & (x <= max(xv)) & (y>=min(yv)) & (y<=max(yv));
 76if ~any(mask)
 77    in = false(inputSize);
 78    on = in;
 79    return
 80end
 81
 82%[xv, yv] = close_loops(xv, yv);
 83Nv = length(xv);
 84
 85% Issue a warning if the bounding box is outside the modeling world that 
 86% we can accurately represent.
 87xrange = max(xv)-min(xv);
 88yrange = max(yv)-min(yv);
 89min_safe_limit = 1.0e-15;
 90max_safe_limit = 1.0e150;
 91
 92
 93
 94inbounds = find(mask);
 95x = x(mask);
 96y = y(mask);
 97
 98
 99% Choose block_length to keep memory usage of vec_inpolygon around
100% 10 Megabytes.
101block_length = 1e5;
102
103M = numel(x);
104
105if M*Nv < block_length
106    if nargout > 1
107        [in on] = vec_inpolygon(Nv,x,y,xv,yv);
108    else
109        in = vec_inpolygon(Nv,x,y,xv,yv);
110    end
111else
112    % Process at most N elements at a time
113    N = ceil(block_length/Nv);
114    in = false(1,M);
115    if nargout > 1
116        on = false(1,M);
117    end
118    n2 = 0;
119    while n2 < M,
120        n1 = n2+1;
121        n2 = n1+N;
122        if n2 > M,
123            n2 = M;
124        end
125        if nargout > 1
126            [in(n1:n2) on(n1:n2)] = vec_inpolygon(Nv,x(n1:n2),y(n1:n2),xv,yv);
127        else
128            in(n1:n2) = vec_inpolygon(Nv,x(n1:n2),y(n1:n2),xv,yv);
129        end
130    end
131end
132
133if nargout > 1
134    onmask = mask;
135    onmask(inbounds(~on)) = 0;
136    on = reshape(onmask, inputSize);
137end
138
139mask(inbounds(~in)) = 0;
140% Reshape output matrix.
141in = reshape(mask, inputSize);
142
143
144%----------------------------------------------
