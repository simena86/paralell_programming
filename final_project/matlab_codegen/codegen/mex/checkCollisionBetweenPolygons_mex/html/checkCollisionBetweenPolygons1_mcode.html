1function bCollide = checkCollisionBetweenPolygons( P1,P2 )
 2% Returns a boolean indicating whether two arbitrary polygons collide.
 3% Requires to polygons defined as n-by-2 arrays of counterclockwise points.
 4% Function first tests whether any vertices of one polygon are inside the
 5% other, then checks for intersection of segments.
 6
 7%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 8% Intro to Robotics, ME170A/ECE181A, Spring 2009
 9% Joey Durham
10% April 21, 2009
11%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
12
13tolerance = 0.001;
14bCollide = false;
15
16% First, check if any vertex of P1 is inside P2
17in2 = inpolygon( P1(:,1), P1(:,2), P2(:,1), P2(:,2) );
18
19if( any(in2) )
20    %fprintf('Vertices in polygon\n');
21    bCollide = true;
22    return
23end
24
25% Next, check if any vertex of P2 is inside P1
26in1 = inpolygon( P2(:,1), P2(:,2), P1(:,1), P1(:,2) );
27
28if( any(in1) )
29    %fprintf('Vertices in polygon\n');
30    bCollide = true;
31    return
32end
33
34% Finally, loop over all combinations of edges of segments checking for
35% intersection.  s1 and s2 are the endpoints of the ith segment of P1, while t1
36% and t2 are for the jth segment of P2
37for i = 1:size(P1,1)
38    s1 = P1(1 + mod(i-1,size(P1,1)), :);
39    s2 = P1(1 + mod(i,size(P1,1)), :);
40    
41    for j = 1:size(P2,1)
42        t1 = P2(1 + mod(j-1,size(P2,1)), :);
43        t2 = P2(1 + mod(j,size(P2,1)), :);
44        
45        den = (t2(2)-t1(2))*(s2(1)-s1(1)) - (t2(1)-t1(1))*(s2(2)-s1(2));
46        
47        u_a = (t2(1)-t1(1))*(s1(2)-t1(2)) - (t2(2)-t1(2))*(s1(1)-t1(1));
48        
49        if( abs(den) < tolerance )
50            if( abs(u_a) < tolerance )
51                % Segments are coincident
52                %fprintf('Coincident segments\n');
53                bCollide = true;
54                return
55            else
56                % Segments are parallel and cannot intersect
57                continue
58            end
59        end
60        
61        u_b = (s2(1)-s1(1))*(s1(2)-t1(2)) - (s2(2)-s1(2))*(s1(1)-t1(1));
62        
63        u_a = u_a / den;
64        u_b = u_b / den;
65        
66        if( u_a >= 0 && u_a <= 1 && u_b >= 0 && u_b <= 1 )
67            % Segments intersect
68            %fprintf('Intersecting segments\n');
69            bCollide = true;
70            return
71        end
72    end
73end
